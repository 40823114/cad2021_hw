#print(data)
#print(len(data))

#print(data)

    num_github = data[i] 
    num = num_github.split("\t")[0]
    account = num_github.split("\t")[1]
    if account == "":
        account = num
    print("<a href='https://" + str(account) + ".github.io/cad2021_hw'>"+ str(num) + "</a>")
